7-  O que imprime o programa abaixo?

        printf("i= %d ", i);
        {
            int i = 30;
            func();
            printf("i= %d ", i);
        }
    }
    void func()
    {
        printf("i = %d ", i);
    }
    

     

    a.  i= 20 i= 20 i= 30 i= 30

        int a = 10, b = 20;
        func(&a, b);
        printf("a = %d, b = %d", a, b);
    }

    b.  a = 20, b = 10
    c.  a = 10, b = 10
